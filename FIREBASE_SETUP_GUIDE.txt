--- RÈGLES DE SÉCURITÉ FIRESTORE ---
Va sur : Firestore > Onglet "Règles"
Remplace tout le contenu par :

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Profils utilisateurs ──────────────────────────────
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // ── Conversations (messagerie patient-médecin) ────────
    match /conversations/{conversationId} {
      allow read: if request.auth != null &&
        (resource.data.patientId == request.auth.uid ||
         resource.data.medecinId == request.auth.uid);
      allow create: if request.auth != null;
      allow update: if request.auth != null &&
        (resource.data.patientId == request.auth.uid ||
         resource.data.medecinId == request.auth.uid);

      // Messages dans une conversation
      match /messages/{messageId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null;
      }
    }

    // ── Repas analysés par ROLLY ──────────────────────────
    match /users/{userId}/repas/{repasId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}

=== ÉTAPE 9 (OPTIONNEL) : PRÉPARER LE RELEASE ===

Pour publier sur le Play Store :

1. Crée une keystore de production :
   keytool -genkey -v -keystore diasmart-release.jks -keyalg RSA -keysize 2048 -validity 10000 -alias diasmart

2. Configure dans app/build.gradle.kts :
   signingConfigs {
       create("release") {
           storeFile = file("diasmart-release.jks")
           storePassword = "ton_mot_de_passe"
           keyAlias = "diasmart"
           keyPassword = "ton_mot_de_passe"
       }
   }

3. Ajoute le SHA-1 et SHA-256 de la keystore release dans Firebase Console

4. Compile en release :
   ./gradlew assembleRelease

5. L'APK signé sera dans :
   app/build/outputs/apk/release/app-release.apk
